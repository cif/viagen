webpack setup

webpack

npm install --save-dev viagen 
// webpack.config.js
const { setupViagen } = require('viagen/webpack')

module.exports = {
  devServer: {
    setupMiddlewares: (middlewares, devServer) => {
      setupViagen(devServer)
      return middlewares
    }
  }
}
setupViagen(devServer, {
  position: 'bottom-right',  // toggle button position
  model: 'sonnet',           // claude model
  panelWidth: 420,           // chat panel width in px
  ui: true,                  // inject chat panel into pages
  systemPrompt: '...',       // custom system prompt
})

Setup & Auth

Once installed, run the setup wizard to authenticate with Claude and write your local .env:

npx viagen setup

Then start your dev server as usual. The viagen chat panel and all /via/* endpoints will be available.
